def create_or_find_format_by name, variant=:default
  fs = @formats[name]
  fs = @formats[name] = {} if fs.nil?
  f = fs[variant]
  if f.nil? 
    f = Workbook::Format.new 
    if variant != :default and fs[:default]
      f = fs[:default].clone
    end
    @formats[name][variant] = f
  end
  return @formats[name][variant]
end
